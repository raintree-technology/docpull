# Pinned release tooling for .github/workflows/publish.yml.
#
# The publish job builds and signs the wheel that is pushed to PyPI via OIDC.
# Installing pip/build/twine with `--upgrade` would pull whatever is *latest* at
# release time, so a single compromised release of any of these packages could
# inject code into the artifact before it is published. Pinning to exact versions
# removes that auto-latest vector. Dependabot/Renovate watch root requirements*.txt
# files and will open update PRs.
pip==26.1.2
build==1.5.0
twine==6.2.0
